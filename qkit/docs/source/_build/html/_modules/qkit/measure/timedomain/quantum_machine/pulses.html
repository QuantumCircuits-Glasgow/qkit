[docs]def gauss(amplitude, delta_t, sigma, length):

    t = np.linspace(-length / 2, length / 2, length)
    gauss_wave = amplitude * np.exp(-((t - delta_t) ** 2) / (2 * sigma ** 2))

    return gauss_wave
