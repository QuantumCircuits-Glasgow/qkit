[docs]    def get_trace(self, tracenumber=1):
        tmp = self._visainstrument.query_ascii_values(':trac:data? trace%i'%tracenumber)
        if self._comp_mode:
            tmp = [tmp,numpy.zeros_like(tmp)]
        return tmp
